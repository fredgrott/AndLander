// This rule specifies where to place the field
// declarations within in the class.
// There are 2 choices:
// 1. Field Declarations before Method Declarations
// 2. Field Declarations after Method Declarations

// Option 1
public class Temperature
{
    private double degreeC;

    public void setDegreeC(double c) {degreeC = c;}
    public double  getDegreeC() {return degreeC;}
}

// Option 2
public class Temperature
{
    public void setDegreeC(double c) {degreeC = c;}
    public double  getDegreeC() {return degreeC;}

    private double degreeC;
}
